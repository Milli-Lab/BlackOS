#blackosdipslay

cmake_minimum_required( VERSION 3.14 FATAL_ERROR)

project (BlackOSDisplay LANGUAGES CXX)

set(SRC_FILES
    src/Display.cpp
    src/Directives.cpp
    src/Kfield.cpp
    src/Kmenu.cpp)

set(INC_FILES
    inc/Directives.h
    inc/Kfield.h
    inc/Kgrid.h
    inc/Kmenu.h
    inc/Kwindow.h)

add_library(Display 
    STATIC
    ${SRC_FILES}
    ${INC_FILES})

add_executable(BlackOSDisplay "src/Display.cpp")


set(CMAKE_CXX_FLAGS "-lncurses")

add_subdirectory(inc)
add_subdirectory(src)
